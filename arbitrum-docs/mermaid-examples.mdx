## Example 2

``` mermaid
%%{
  init: {
    "theme": "base",
    "flowchart" : { "curve" : "stepAfter", "rankSpacing" : "80","wrappingWidth" :"800" },
    "themeVariables": {
      "primaryColor": "#BB2528",
      "primaryTextColor": "#fff",
      "primaryBorderColor": "#7C0000",
      "lineColor": "#F8B229",
      "secondaryColor": "#006100",
      "tertiaryColor": "#fff"
    }
  }
}%%
block-beta
  columns 6
  block:L3SettlingOnEthereumL2s:2
    1("Orbit L3"):1
    2("Orbit L3"):1 
    end
  block:L3SettlingOnArbitrumL2s:4
    3("Orbit L3"):1 
    4("Orbit L3"):1 
    5("Orbit L3"):1 
    6("Orbit L3"):1 
    end
  block:L2s:6
  ETHL2("Ethereum L2"):2
  ArbOneArbNov("Arbitrum One/Nova"):3
  OrbitL2("Orbit L2"):1
  end
  EthereumL1("Ethereum L1"):6
  1-->ETHL2
  2-->ETHL2
  3-->ArbOneArbNov
  4-->ArbOneArbNov
  5-->ArbOneArbNov
  6-->ArbOneArbNov
  ETHL2-->EthereumL1
  ArbOneArbNov-->EthereumL1
  OrbitL2-->EthereumL1
  style L2s fill:#636,stroke:#333,stroke-width:4px
```
